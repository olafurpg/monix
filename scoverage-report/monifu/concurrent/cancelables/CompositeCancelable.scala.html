24 /**
25  * Represents a composite of multiple cancelables. In case it is canceled, all
26  * contained cancelables will be canceled too, e.g...
27  * {{{
28  *   val s = CompositeCancelable()
29  *
30  *   s += c1
31  *   s += c2
32  *   s += c3
33  *
34  *   // c1, c2, c3 will also be canceled
35  *   s.cancel()
36  * }}}
37  *
38  * Additionally, once canceled, on appending of new cancelable references, those
39  * references will automatically get canceled too:
40  * {{{
41  *   val s = CompositeCancelable()
42  *   s.cancel()
43  *
44  *   // c1 gets canceled, because s is already canceled
45  *   s += c1
46  *   // c2 gets canceled, because s is already canceled
47  *   s += c2
48  * }}}
49  *
50  * Adding and removing references from this composite is thread-safe.
51  */
52 trait CompositeCancelable extends BooleanCancelable {
53   /**
54    * Adds a cancelable reference to this composite.
55    */
56   def add(s: Cancelable): Unit
57 
58   /**
59    * Adds a cancelable reference to this composite.
60    * This is an alias for `add()`.
61    */
62   final def +=(other: Cancelable): Unit = add(other)
63 
64   /**
65    * Removes a cancelable reference from this composite.
66    */
67   def remove(s: Cancelable): Unit
68 
69   /**
70    * Removes a cancelable reference from this composite.
71    * This is an alias for `remove()`.
72    */
73   final def -=(s: Cancelable): Unit = remove(s)
74 }
75 
76 object CompositeCancelable {
77   def apply(): CompositeCancelable =
78     new CompositeCancelableImpl
79 
80   def apply(head: Cancelable, tail: Cancelable*): CompositeCancelable = {
81     val cs = new CompositeCancelableImpl
82     cs += head; for (os <- tail) cs += os
83     cs
84   }
85 
86   private[this] final class CompositeCancelableImpl extends CompositeCancelable {
87     def isCanceled =
88       state.get.isCanceled
89 
90     @tailrec
91     def cancel(): Boolean = {
92       val oldState = state.get
93       if (!oldState.isCanceled)
94         if (!state.compareAndSet(oldState, State(Set.empty, isCanceled = true)))
95           cancel()
96         else {
97           for (s <- oldState.subscriptions) s.cancel()
98           true
99         }
100       else
101         false
102     }
103 
104     @tailrec
105     def add(s: Cancelable): Unit = {
106       val oldState = state.get
107       if (oldState.isCanceled)
108         s.cancel()
109       else {
110         val newState = oldState.copy(subscriptions = oldState.subscriptions + s)
111         if (!state.compareAndSet(oldState, newState))
112           add(s)
113       }
114     }
115 
116     @tailrec
117     def remove(s: Cancelable): Unit = {
118       val oldState = state.get
119       if (!oldState.isCanceled) {
120         val newState = oldState.copy(subscriptions = oldState.subscriptions - s)
121         if (!state.compareAndSet(oldState, newState))
122           remove(s)
123       }
124     }
125 
126     private[this] val state: AtomicAny[State] =
127       Atomic(State())
128 
129     private[this] case class State(
130       subscriptions: Set[Cancelable] = Set.empty,
131       isCanceled: Boolean = false
132     )
133   }
134 }
