24 /**
25  * Represents the acknowledgement of processing that a consumer
26  * sends back upstream on `Observer.onNext`
27  */
28 sealed trait Ack extends Future[Ack]
29 
30 object Ack {
31   /**
32    * Acknowledgement of processing that signals upstream that the
33    * consumer is interested in receiving more events.
34    */
35   sealed trait Continue extends Ack
36 
37   case object Continue extends Continue with AckIsFuture[Continue] {
38     final val IsSuccess = Success(Continue)
39     final val value = Some(IsSuccess)
40   }
41 
42   /**
43    * Acknowledgement or processing that signals upstream that the
44    * consumer is no longer interested in receiving events.
45    */
46   sealed trait Cancel extends Ack
47 
48   case object Cancel extends Cancel with AckIsFuture[Cancel] {
49     final val IsSuccess = Success(Cancel)
50     final val value = Some(IsSuccess)
51   }
52 }
53 
54 sealed trait AckIsFuture[T <: Ack] extends Future[T] { self =>
55   final val isCompleted = true
56 
57   final def ready(atMost: Duration)(implicit permit: CanAwait) = self
58   final def result(atMost: Duration)(implicit permit: CanAwait) = value.get.get
59 
60   final def onComplete[U](func: (Try[T]) => U)(implicit executor: ExecutionContext): Unit =
61     executor.execute(new Runnable {
62       def run(): Unit = func(value.get)
63     })
