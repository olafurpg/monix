34   def left[T](source: Observable[T], n: Int): Observable[T] =
35     Observable.create { subscriber =>
36       implicit val s = subscriber.scheduler
37       val observer = subscriber.observer
38 
39       source.unsafeSubscribe(new Observer[T] {
40         private[this] var counter = 0
41         private[this] var isDone = false
42 
43         def onNext(elem: T) = {
44           if (n <= 0 && !isDone) {
45             isDone = true
46             observer.onComplete()
47             Cancel
48           }
49           else if (!isDone && counter < n) {
50             // ^^ short-circuit for not endlessly incrementing that number
51             counter += 1
52 
53             if (counter < n) {
54               // this is not the last event in the stream, so send it directly
55               observer.onNext(elem)
56             }
57             else  {
58               // last event in the stream, so we need to send the event followed by an EOF downstream
59               // after which we signal upstream to the producer that it should stop
60               isDone = true
61               observer.onNext(elem)
62                 .onContinueSignalComplete(observer)
63               Cancel
64             }
65           }
66           else {
67             // we already emitted the maximum number of events, so signal upstream
68             // to the producer that it should stop sending events
69             Cancel
70           }
71         }
72 
73         def onError(ex: Throwable) =
74           if (!isDone) {
75             isDone = true
76             observer.onError(ex)
77           }
78 
79         def onComplete() =
80           if (!isDone) {
81             isDone = true
82             observer.onComplete()
83           }
84       })
85     }

88    * Implementation for [[monifu.reactive.Observable.takeRight]].
89    */
90   def right[T](source: Observable[T], n: Int): Observable[T] =
91     Observable.create { subscriber =>
92       implicit val s = subscriber.scheduler
93       val observer = subscriber.observer
94 
95       source.unsafeSubscribe(new Observer[T] {
96         private[this] val queue = mutable.Queue.empty[T]
97         private[this] var queued = 0
98 
99         def onNext(elem: T): Future[Ack] = {
100           if (n <= 0)
101             Cancel
102           else if (queued < n) {
103             queue.enqueue(elem)
104             queued += 1
105           }
106           else {
107             queue.enqueue(elem)
108             queue.dequeue()
109           }
110           Continue
111         }
112 
113         def onComplete(): Unit = {
114           Observable.fromIterable(queue).unsafeSubscribe(observer)
115         }
116 
117         def onError(ex: Throwable): Unit = {
118           Observable.fromIterable(queue).unsafeSubscribe(new Observer[T] {
119             def onError(ex: Throwable) =
120               observer.onError(ex)
121 
122             def onComplete() =
123               observer.onError(ex)
124 
125             def onNext(elem: T) =
126               observer.onNext(elem)
127           })
128         }
129       })
130     }

132   def leftByTimespan[T](source: Observable[T], timespan: FiniteDuration): Observable[T] =
133     Observable.create { subscriber =>
134       implicit val s = subscriber.scheduler
135       val observer = subscriber.observer
136 
137       source.unsafeSubscribe(new Observer[T] with Runnable {
138         private[this] var isActive = true
139         private[this] val task = s.scheduleOnce(timespan, this)
140 
141         def onNext(elem: T): Future[Ack] = synchronized {
142           if (isActive)
143             observer.onNext(elem)
144               .ifCanceledDoCancel(task)
145           else {
146             onComplete()
147             Cancel
148           }
149         }
150 
151         def onError(ex: Throwable): Unit = synchronized {
152           if (isActive) {
153             isActive = false
154             task.cancel()
155             observer.onError(ex)
156           }
157         }
158 
159         def onComplete(): Unit = synchronized {
160           if (isActive) {
161             isActive = false
162             task.cancel()
163             observer.onComplete()
164           }
165         }
166 
167         def run(): Unit = {
168           onComplete()
169         }
170       })
171     }

176   def byPredicate[T](source: Observable[T])(p: T => Boolean): Observable[T] =
177     Observable.create[T] { subscriber =>
178       implicit val s = subscriber.scheduler
179       val observer = subscriber.observer
180 
181       source.unsafeSubscribe(new Observer[T] {
182         var shouldContinue = true
183 
184         def onNext(elem: T) = {
185           if (shouldContinue) {
186             // Protects calls to user code from within an operator
187             var streamError = true
188             try {
189               val isValid = p(elem)
190               streamError = false
191 
192               if (isValid) {
193                 observer.onNext(elem)
194               }
195               else {
196                 shouldContinue = false
197                 observer.onComplete()
198                 Cancel
199               }
200             }
201             catch {
202               case NonFatal(ex) =>
203                 if (streamError) {
204                   observer.onError(ex)
205                   Cancel
206                 }
207                 else
208                   Future.failed(ex)
209             }
210           }
211           else
212             Cancel
213         }
214 
215         def onComplete() = {
216           observer.onComplete()
217         }
218 
219         def onError(ex: Throwable) = {
220           observer.onError(ex)
221         }
222       })
223     }
