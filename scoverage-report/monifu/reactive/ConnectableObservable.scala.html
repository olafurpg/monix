17 package monifu.reactive
18 
19 import monifu.concurrent.Scheduler
20 import monifu.concurrent.cancelables.BooleanCancelable
21 
22 /**
23  * Represents an [[monifu.reactive.Observable Observable]] that waits for
24  * the call to `connect()` before
25  * starting to emit elements to its subscriber(s).
26  *
27  * Useful for converting cold observables into hot observables and thus returned by
28  * [[monifu.reactive.Observable.multicast Observable.multicast]].
29  */
30 trait ConnectableObservable[+T] extends Observable[T] {
31   /**
32    * Starts emitting events to subscribers.
33    */
34   def connect(): BooleanCancelable
35 }
36 
37 
38 object ConnectableObservable {
39   /**
40    * Builds a [[ConnectableObservable]] for the given observable source
41    * and a given [[Subject]].
42    */
43   def apply[T, R](source: Observable[T], subject: Subject[T, R])
44       (implicit s: Scheduler): ConnectableObservable[R] = {
45 
46     new ConnectableObservable[R] {
47       private[this] lazy val connection = {
48         source.subscribe(subject)
49       }
50 
51       def connect() = {
52         connection
53       }
54 
55       def subscribeFn(subscriber: Subscriber[R]): Unit = {
56         subject.unsafeSubscribe(subscriber)
57       }
58     }
59   }
60 }
