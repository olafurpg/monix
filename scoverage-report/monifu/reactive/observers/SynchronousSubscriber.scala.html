17 package monifu.reactive.observers

19 import monifu.concurrent.Scheduler
20 import monifu.reactive.Subscriber
21 
22 /**
23  * A `SynchronousSubscriber` is a [[Subscriber]] whose observer
24  * signals demand synchronously (i.e. the upstream observable doesn't need to
25  * wait on a `Future` in order to decide whether to send the next event
26  * or not).
27  */
28 trait SynchronousSubscriber[-T] extends Subscriber[T] {
29   def observer: SynchronousObserver[T]
30 }
31 
32 object SynchronousSubscriber {
33   /** Subscriber builder */
34   def apply[T](observer: SynchronousObserver[T], scheduler: Scheduler): SynchronousSubscriber[T] =
35     new Implementation[T](observer, scheduler)
36 
37   /** Utility for pattern matching */
38   def unapply[T](ref: SynchronousSubscriber[T]): Some[(SynchronousObserver[T], Scheduler)] =
39     Some((ref.observer, ref.scheduler))
40 
41   private[this] final class Implementation[-T]
42       (val observer: SynchronousObserver[T], val scheduler: Scheduler)
43     extends SynchronousSubscriber[T] {
44 
45     require(observer != null, "Observer should not be null")
46     require(scheduler != null, "Scheduler should not be null")
47 
48     override def equals(other: Any): Boolean = other match {
49       case that: Implementation[_] =>
50         observer == that.observer &&
51           scheduler == that.scheduler
52       case _ =>
53         false
54     }
55 
56     override val hashCode: Int = {
57       31 * observer.hashCode() + scheduler.hashCode()
58     }
59   }}
