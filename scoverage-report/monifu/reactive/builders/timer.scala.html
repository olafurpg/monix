25 object timer {
26   /**
27    * Create an Observable that repeatedly emits the given `item`, until
28    * the underlying Observer cancels.
29    */
30   def repeated[T](initialDelay: FiniteDuration, period: FiniteDuration, unit: T): Observable[T] = {
31     Observable.create { subscriber =>
32       implicit val s = subscriber.scheduler
33       val observer = subscriber.observer
34 
35       // we are deploying optimizations in order to reduce garbage allocations,
36       // therefore the weird Runnable instance that does a state machine
37       val runnable = new Runnable { self =>
38         import ObserverState.{ON_CONTINUE, ON_NEXT}
39 
40         private[this] val periodMs = period.toMillis
41         private[this] var state = ON_NEXT
42         private[this] var startedAt = 0L
43 
44         def run(): Unit = state match {
45           case ON_NEXT =>
46             state = ON_CONTINUE
47             startedAt = s.currentTimeMillis()
48             observer.onNext(unit).onContinue(self)
49 
50           case ON_CONTINUE =>
51             state = ON_NEXT
52             val initialDelay = {
53               val duration = s.currentTimeMillis() - startedAt
54               val d = periodMs - duration
55               if (d >= 0L) d else 0L
56             }
57 
58             s.scheduleOnce(initialDelay, TimeUnit.MILLISECONDS, self)
59         }
60       }
61 
62       s.scheduleOnce(initialDelay, runnable)
63     }
64   }
65 }
