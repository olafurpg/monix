32     /**
33      * If the result of this `Future` is a `Continue`, then completes
34      * the stream with an `onComplete`.
35      *
36      * Note that it doesn't do exception handling - if the future is a failure
37      * then the behavior is undefined.
38      */
39     def onContinueSignalComplete[T](observer: Observer[T])(implicit s: Scheduler): Unit =
40       source match {
41         case sync if sync.isCompleted =>
42           sync.value.get match {
43             case Continue.IsSuccess =>
44               observer.onComplete()
45             case _ =>
46               () // do nothing
47           }
48 
49         case async =>
50           async.onComplete {
51             case Continue.IsSuccess =>
52               observer.onComplete()
53             case _ =>
54               () // do nothing
55           }
56       }

83     /**
84      * On Continue, triggers the execution of the given callback.
85      */
86     def onContinue(callback: => Unit)(implicit s: Scheduler): Unit =
87       source match {
88         case sync if sync.isCompleted =>
89           if (sync == Continue || (sync != Cancel && sync.value.get == Continue.IsSuccess))
90             try callback catch {
91               case NonFatal(ex) =>
92                 s.reportFailure(ex)
93             }
94 
95         case async =>
96           async.onComplete {
97             case Continue.IsSuccess =>
98               try callback catch {
99                 case NonFatal(ex) =>
100                   s.reportFailure(ex)
101               }
102             case _ =>
103               () // do nothing
104           }
105       }

136     /**
137      * Helper that triggers [[Cancelable.cancel]] on the given reference,
138      * in case our source is a [[Cancel]] or a failure.
139      */
140     def ifCanceledDoCancel(cancelable: Cancelable)(implicit s: Scheduler): Future[Ack] = {
141       source match {
142         case sync if sync.isCompleted =>
143           sync.value.get match {
144             case Cancel.IsSuccess | Failure(_) =>
145               cancelable.cancel()
146               sync
147             case _ =>
148               sync
149           }
150 
151         case async =>
152           async.onComplete {
153             case Cancel.IsSuccess | Failure(_) =>
154               cancelable.cancel()
155             case _ =>
156               ()
157           }
158           async
159       }

162     /**
163      * An implementation of `flatMap` that executes
164      * synchronously if the source is already completed.
165      */
166     def fastFlatMap[R](f: Ack => Future[R])(implicit ec: ExecutionContext): Future[R] =
167       source.value match {
168         case Some(sync) =>
169           sync match {
170             case Success(ack) =>
171               try f(ack) catch {
172                 case NonFatal(ex) =>
173                   Future.failed(ex)
174               }
175             case Failure(_) =>
176               source.asInstanceOf[Future[R]]
177           }
178         case None =>
179           source.flatMap(f)
180       }

184     def onContinueStreamOnNext[T](observer: Observer[T], nextElem: T)(implicit s: Scheduler) =
185       source match {
186         case sync if sync.isCompleted =>
187           if (sync == Continue || sync.value.get == Continue.IsSuccess)
188             observer.onNext(nextElem)
189           else
190             Cancel
191         case async =>
192           async.flatMap {
193             case Continue =>
194               observer.onNext(nextElem)
195             case Cancel =>
196               Cancel
197           }
198       }

200     def onContinueCompleteWith[T](observer: Observer[T], lastElem: T)(implicit s: Scheduler): Unit =
201       source match {
202         case sync if sync.isCompleted =>
203           if (sync == Continue || ((sync != Cancel) && sync.value.get == Continue.IsSuccess)) {
204             try {
205               observer.onNext(lastElem)
206               observer.onComplete()
207             }
208             catch {
209               case NonFatal(err) =>
210                 observer.onError(err)
211             }
212           }
213         case async =>
214           async.onSuccess {
215             case Continue =>
216               try {
217                 observer.onNext(lastElem)
218                 observer.onComplete()
219               }
220               catch {
221                 case NonFatal(err) =>
222                   observer.onError(err)
223               }
224           }
225       }

230     def onCancelContinue(p: Promise[Ack])(implicit s: Scheduler): Future[Ack] = {
231       source match {

233         case Cancel => p.success(Continue)
234 
235         case sync if sync.isCompleted && sync.value.get.isSuccess =>
236           sync.value.get.get match {
237             case Continue => // do nothing

240 
241         case async =>

246             case other =>
247               // branch not necessary, but Scala's compiler emits warnings if missing
248               s.reportFailure(new MatchError(other.toString))

316     /**
317      * Unsafe version of `onComplete` that triggers execution synchronously
318      * in case the source is already completed.
319      */
320     def onCompleteNow(f: Try[Ack] => Unit)(implicit s: Scheduler): Future[Ack] =
321       source match {
322         case sync if sync.isCompleted =>
323           try f(sync.value.get) catch {
324             case NonFatal(ex) =>

326           }
327           source
