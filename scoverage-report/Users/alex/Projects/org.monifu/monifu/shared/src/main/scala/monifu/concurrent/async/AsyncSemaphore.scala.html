25 final class AsyncSemaphore private (limit: Int) {
26   private[this] val queue = {
27     val elems = for (i <- 0 until limit) yield ()
28     AsyncQueue(elems :_*)
29   }
30 
31   def acquire[T](timeout: FiniteDuration)(f: => Future[T])(implicit s: Scheduler): Future[T] =
32     queue.poll().flatMap { _ =>
33       val t = f.withTimeout(timeout)
34       t.onComplete { _ => queue.offer(()) }
35       t
36     }
37 
38   def acquire[T](f: => Future[T])(implicit ec: ExecutionContext): Future[T] =
39     queue.poll().flatMap { _ =>
40       f.onComplete { _ => queue.offer(()) }

42     }
43 }
