40   private[this] val queue = ConcurrentQueue.empty[T]
41   // to be modified only in onError, before upstreamIsComplete
42   private[this] var errorThrown: Throwable = null
43   // to be modified only in onError / onComplete
44   @volatile private[this] var upstreamIsComplete = false
45   // to be modified only by consumer
46   @volatile private[this] var downstreamIsDone = false

48   // for enforcing non-concurrent updates and back-pressure
49   // all access must be synchronized
50   private[this] val lock = new AnyRef
51   private[this] var itemsToPush = 0
52   private[this] var nextAckPromise = Promise[Ack]()
53   private[this] var appliesBackPressure = false

55   val observer: Observer[T] = new Observer[T] {
56     def onNext(elem: T): Future[Ack] = lock.synchronized {
57       if (!upstreamIsComplete && !downstreamIsDone) {
58         try {
59           queue.offer(elem)
60           pushToConsumer()
61         }
62         catch {
63           case NonFatal(ex) =>
64             onError(ex)
65             Cancel
66         }
67       }
68       else {
69         Cancel
70       }
71     }
72 
73     def onError(ex: Throwable) = lock.synchronized {
74       if (!upstreamIsComplete && !downstreamIsDone) {
75         errorThrown = ex
76         upstreamIsComplete = true
77         pushToConsumer()
78       }
79     }
80 
81     def onComplete() = lock.synchronized {
82       if (!upstreamIsComplete && !downstreamIsDone) {
83         upstreamIsComplete = true
84         pushToConsumer()
85       }
86     }
87   }

89   private[this] def pushToConsumer(): Future[Ack] = {
90     if (itemsToPush == 0) {
91       nextAckPromise = Promise[Ack]()
92       appliesBackPressure = false
93       itemsToPush += 1
94 
95       scheduler.execute(new Runnable {
96         def run() = fastLoop(0)
97       })
98 
99       Continue
100     }
101     else if (appliesBackPressure) {
102       itemsToPush += 1
103       nextAckPromise.future
104     }
105     else if (itemsToPush >= bufferSize) {
106       appliesBackPressure = true
107       itemsToPush += 1
108       nextAckPromise.future
109     }
110     else {
111       itemsToPush += 1
112       Continue
113     }
114   }

116   private[this] def rescheduled(processed: Int): Unit = {
117     fastLoop(processed)
118   }
119 
120   @tailrec
121   private[this] def fastLoop(processed: Int): Unit = {
122     if (!downstreamIsDone) {
123       val hasError = errorThrown ne null
124       val next: T = queue.poll()
125 
126       if (next != null)
127         underlying.onNext(next) match {
128           case sync if sync.isCompleted =>
129             sync match {
130               case continue if continue == Continue || continue.value.get == Continue.IsSuccess =>
131                 // process next
132                 fastLoop(processed + 1)
133 
134               case done if done == Cancel || done.value.get == Cancel.IsSuccess =>
135                 // ending loop
136                 downstreamIsDone = true
137                 lock.synchronized {
138                   itemsToPush = 0
139                   nextAckPromise.success(Cancel)
140                 }
141 
142               case error if error.value.get.isFailure =>
143                 // ending loop
144                 downstreamIsDone = true
145                 try underlying.onError(error.value.get.failed.get) finally
146                   lock.synchronized {
147                     itemsToPush = 0
148                     nextAckPromise.success(Cancel)
149                   }
150             }
151 
152           case async =>
153             async.onComplete {
154               case Continue.IsSuccess =>
155                 // re-run loop (in different thread)
156                 rescheduled(processed + 1)
157 
158               case Cancel.IsSuccess =>
159                 // ending loop
160                 downstreamIsDone = true
161                 lock.synchronized {
162                   itemsToPush = 0
163                   nextAckPromise.success(Cancel)
164                 }
165 
166               case Failure(ex) =>
167                 // ending loop
168                 downstreamIsDone = true
169                 try underlying.onError(ex) finally
170                   lock.synchronized {
171                     itemsToPush = 0
172                     nextAckPromise.success(Cancel)
173                   }
174 
175               case other =>
176                 // never happens, but to appease Scala's compiler
177                 downstreamIsDone = true
178                 try underlying.onError(new MatchError(s"$other")) finally
179                   lock.synchronized {
180                     itemsToPush = 0
181                     nextAckPromise.success(Cancel)
182                   }
183             }
184         }
185       else if (upstreamIsComplete || hasError) {
186         // Race-condition check, but if upstreamIsComplete=true is visible, then the queue should be fully published
187         // because there's a clear happens-before relationship between queue.offer() and upstreamIsComplete=true
188         // NOTE: errors have priority, so in case of an error seen, then the loop is stopped
189         if (!queue.isEmpty) {
190           fastLoop(processed) // re-run loop
191         }
192         else {
193           // ending loop
194           downstreamIsDone = true
195           try {
196             if (errorThrown ne null)
197               underlying.onError(errorThrown)
198             else
199               underlying.onComplete()
200           }
201           finally lock.synchronized {
202             queue.clear() // for GC purposes
203             itemsToPush = 0
204             nextAckPromise.success(Cancel)
205           }
206         }
207       }
208       else {
209         val remaining = lock.synchronized {
210           itemsToPush -= processed
211           if (itemsToPush <= 0) // this really has to be LESS-or-equal
212             nextAckPromise.success(Continue)
213           itemsToPush
214         }
215 
216         // if the queue is non-empty (i.e. concurrent modifications might have happened)
217         // then start all over again
218         if (remaining > 0) fastLoop(0)
219       }
220     }
221   }
