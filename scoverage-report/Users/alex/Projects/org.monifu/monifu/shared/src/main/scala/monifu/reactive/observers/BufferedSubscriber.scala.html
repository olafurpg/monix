23 /**
24  * Interface describing [[monifu.reactive.Observer Observer]] wrappers
25  * that are thread-safe (can receive concurrent events) and that
26  * return an immediate `Continue` when receiving `onNext`
27  * events. Meant to be used by data sources that cannot uphold the
28  * no-concurrent events and the back-pressure related requirements
29  * (i.e. data-sources that cannot wait on `Future[Continue]` for
30  * sending the next event).
31  *
32  * Implementations of this interface have the following contract:
33  *
34  *  - `onNext` / `onError` / `onComplete` of this interface MAY be
35  *    called concurrently
36  *  - `onNext` SHOULD return an immediate `Continue`, as long as the
37  *    buffer is not full and the underlying observer hasn't signaled
38  *    `Cancel` (N.B. due to the asynchronous nature, `Cancel` signaled
39  *    by the underlying observer may be noticed later, so
40  *    implementations of this interface make no guarantee about queued
41  *    events - which could be generated, queued and dropped on the
42  *    floor later)
43  *  - `onNext` MUST return an immediate `Cancel` result, after it
44  *    notices that the underlying observer signaled `Cancel` (due to
45  *    the asynchronous nature of observers, this may happen later and
46  *    queued events might get dropped on the floor)
47  *  - in general the contract for the underlying Observer is fully
48  *    respected (grammar, non-concurrent notifications, etc...)
49  *  - when the underlying observer canceled (by returning `Cancel`),
50  *    or when a concurrent upstream data source triggered an error,
51  *    this SHOULD eventually be noticed and acted upon
52  *  - as long as the buffer isn't full and the underlying observer
53  *    isn't `Cancel`, then implementations of this interface SHOULD
54  *    not lose events in the process
55  *  - the buffer MAY BE either unbounded or bounded, in case of
56  *    bounded buffers, then an appropriate policy needs to be set for
57  *    when the buffer overflows - either an `onError` triggered in the
58  *    underlying observer coupled with a `Cancel` signaled to the
59  *    upstream data sources, or dropping events from the head or the
60  *    tail of the queue, or attempting to apply back-pressure, etc...
61  *
62  * See [[monifu.reactive.BufferPolicy BufferPolicy]] for the buffer
63  * policies available.
64  */
65 trait BufferedSubscriber[-T] extends Subscriber[T]
66 
67 object BufferedSubscriber {
68   def apply[T](observer: Observer[T], bufferPolicy: BufferPolicy[T])
69       (implicit s: Scheduler): BufferedSubscriber[T] = {
70 
71     bufferPolicy match {
72       case Unbounded =>
73         SynchronousBufferedSubscriber.unbounded(observer)
74       case OverflowTriggering(bufferSize) =>
75         SynchronousBufferedSubscriber.overflowTriggering(observer, bufferSize)
76       case BackPressured(bufferSize) =>
77         BackPressuredBufferedSubscriber(observer, bufferSize)
78       case DropIncoming(bufferSize) =>
79         DropIncomingBufferedSubscriber.simple(observer, bufferSize)
80       case DropIncomingThenSignal(bufferSize, onOverflow) =>
81         DropIncomingBufferedSubscriber.withSignal(observer, bufferSize, onOverflow)
82     }
83   }
84 }
