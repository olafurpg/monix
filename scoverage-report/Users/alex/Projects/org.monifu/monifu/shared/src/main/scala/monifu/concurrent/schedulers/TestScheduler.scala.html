31  * A scheduler meant for testing purposes.
32  */
33 final class TestScheduler private () extends ReferenceScheduler {
34   /*
35    * The `internalClock` is used for executing tasks. Upon calling [[tick]], the
36    * internal clock is advanced and pending `tasks` are executed.
37    */
38   val state = Atomic(State(
39     lastID = 0,
40     clock = Duration.Zero,
41     tasks = SortedSet.empty[Task],
42     cancelTask = cancelTask,
43     lastReportedError = null
44   ))
45 
46   override def currentTimeMillis(): Long =
47     state.get.clock.toMillis
48 
49   def scheduleOnce(initialDelay: FiniteDuration, r: Runnable): Cancelable =
50     state.transformAndExtract(_.scheduleOnce(initialDelay, r))
51 
52   def scheduleOnce(initialDelay: Long, unit: TimeUnit, r: Runnable): Cancelable =
53     state.transformAndExtract(_.scheduleOnce(FiniteDuration(initialDelay, unit), r))
54 
55   private[this] def cancelTask(t: Task): Unit =
56     state.transform(s => s.copy(tasks = s.tasks - t))
57 
58   def execute(runnable: Runnable): Unit = {
59     state.transform(_.execute(runnable))
60   }
61 
62   def reportFailure(t: Throwable): Unit = {
63     state.transform(_.copy(lastReportedError = t))
64   }
65 
66   private[this] def extractOneTask(current: State, clock: FiniteDuration) = {
67     current.tasks.headOption.filter(_.runsAt <= clock) match {
68       case Some(value) =>
69         val firstTick = value.runsAt
70         val immediateT = current.tasks.takeWhile(_.runsAt == firstTick)
71         val shuffled = Random.shuffle(immediateT.toVector)
72 
73         val forExecution = shuffled.head
74         val remaining = (current.tasks -- immediateT) ++ shuffled.drop(1)
75 
76         assert(!remaining.contains(forExecution), "contract breach")
77         assert(remaining.size == current.tasks.size - 1, "contract breach")
78 
79         Some((forExecution, remaining))
80 
81       case None =>
82         None
83     }
84   }
85 
86   @tailrec
87   def tickOne(): Boolean = {
88     val current = state.get
89 
90     // extracting one task by taking the immediate tasks
91     extractOneTask(current, current.clock) match {
92       case Some((head, rest)) =>
93         if (!state.compareAndSet(current, current.copy(tasks = rest)))
94           tickOne()
95         else {
96           // execute task
97           try head.task.run() catch {
98             case NonFatal(ex) =>
99               reportFailure(ex)
100           }
101 
102           true
103         }
104 
105       case None =>
106         false
107     }
108   }
109 
110   def tick(time: FiniteDuration = Duration.Zero): Boolean = {
111     @tailrec
112     def loop(time: FiniteDuration, result: Boolean): Boolean = {
113       val current = state.get
114       val currentClock = current.clock + time
115 
116       extractOneTask(current, currentClock) match {
117         case Some((head, rest)) =>
118           if (!state.compareAndSet(current, current.copy(clock = head.runsAt, tasks = rest)))
119             loop(time, result)
120           else {
121             // execute task
122             try head.task.run() catch {
123               case NonFatal(ex) =>
124                 reportFailure(ex)
125             }
126 
127             // have to retry execution, as those pending tasks
128             // may have registered new tasks for immediate execution
129             loop(currentClock - head.runsAt, result = true)
130           }
131 
132         case None =>
133           if (!state.compareAndSet(current, current.copy(clock = currentClock)))
134             loop(time, result)
135           else
136             result
137       }
138     }
139 
140     loop(time, result = false)
141   }
142 }

144 object TestScheduler {
145   /**
146    * Builder for [[TestScheduler]].
147    */
148   def apply(): TestScheduler = {

152   /**
153    * Used internally by [[TestScheduler]], represents a
154    * unit of work pending execution.
155    */
156   case class Task(id: Long, task: Runnable, runsAt: FiniteDuration)
157 
158   object Task {
159     implicit val ordering: Ordering[Task] =
160       new Ordering[Task] {
161         val longOrd = implicitly[Ordering[Long]]
162 
163         def compare(x: Task, y: Task): Int =
164           x.runsAt.compare(y.runsAt) match {
165             case nonZero if nonZero != 0 =>
166               nonZero
167             case _ =>
168               longOrd.compare(x.id, y.id)
169           }
170       }
171   }
172 
173   /**
174    * Used internally by [[TestScheduler]], represents the internal
175    * state used for task scheduling and execution.
176    */
177   case class State(
178     lastID: Long,
179     clock: FiniteDuration,
180     tasks: SortedSet[Task],
181     cancelTask: Task => Unit,
182     lastReportedError: Throwable) {
183 
184     assert(!tasks.headOption.exists(_.runsAt < clock),
185       "The runsAt for any task must never be in the past")
186 
187     /**
188      * Returns a new state with the runnable scheduled for execution.
189      */
190     def execute(runnable: Runnable): State = {
191       val newID = lastID + 1
192       val task = Task(newID, runnable, clock)
193       copy(lastID = newID, tasks = tasks + task)
194     }
195 
196     /**
197      * Returns a new state with a scheduled task included.
198      */
199     def scheduleOnce(delay: FiniteDuration, r: Runnable): (Cancelable, State) = {
200       require(delay >= Duration.Zero, "The given delay must be positive")
201 
202       val newID = lastID + 1
203       val cancelable = SingleAssignmentCancelable()
204 
205       val task = Task(newID, r, this.clock + delay)
206       cancelable := Cancelable { cancelTask(task) }
207 
208       (cancelable, copy(
209         lastID = newID,
210         tasks = tasks + task
211       ))
212     }
213   }
