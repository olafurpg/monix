95   def byPredicate[T](source: Observable[T])(p: T => Boolean): Observable[T] =
96     Observable.create { subscriber =>
97       implicit val s = subscriber.scheduler
98       val observer = subscriber.observer
99 
100       source.unsafeSubscribe(new Observer[T] {
101         var continueDropping = true
102 
103         def onNext(elem: T) = {
104           if (continueDropping) {
105             // See Section 6.4. in the Rx Design Guidelines:
106             // Protect calls to user code from within an operator
107             var streamError = true
108             try {
109               val isStillInvalid = p(elem)
110               streamError = false
111 
112               if (isStillInvalid)
113                 Continue
114               else {
115                 continueDropping = false
116                 observer.onNext(elem)
117               }
118             }
119             catch {
120               case NonFatal(ex) =>
121                 if (streamError) { observer.onError(ex); Cancel } else Future.failed(ex)
122             }
123           }
124           else
125             observer.onNext(elem)
126         }
127 
128         def onComplete() =
129           observer.onComplete()
130 
131         def onError(ex: Throwable) =
132           observer.onError(ex)
133       })
134     }
