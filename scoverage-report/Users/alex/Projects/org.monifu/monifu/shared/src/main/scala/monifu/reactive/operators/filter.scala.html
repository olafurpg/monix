33       source.unsafeSubscribe(new Observer[T] {
34         def onNext(elem: T) = {
35           // Protects calls to user code from within the operator
36           var streamError = true
37           try {
38             if (p(elem)) {
39               streamError = false
40               observer.onNext(elem)
