64   /**
65    * Implementation for [[monifu.reactive.Observable.defaultIfEmpty]].
66    */
67   def defaultIfEmpty[T](source: Observable[T], default: T): Observable[T] =
68     Observable.create { subscriber =>
69       implicit val s = subscriber.scheduler
70       val observer = subscriber.observer
71 
72       source.unsafeSubscribe(new Observer[T] {
73         private[this] var isEmpty = true
74 
75         def onNext(elem: T): Future[Ack] = {
76           if (isEmpty) isEmpty = false
77           observer.onNext(elem)
78         }
79 
80         def onError(ex: Throwable): Unit = {
81           observer.onError(ex)
82         }
83 
84         def onComplete(): Unit = {
85           if (isEmpty)
86             observer.onNext(default)
87               .onContinueSignalComplete(observer)
88           else
89             observer.onComplete()
90         }
91       })
92     }
93 
94   /**
95    * Implements [[Observable.endWithError]].
96    */
97   def endWithError[T](source: Observable[T])(error: Throwable): Observable[T] =
98     Observable.create { subscriber =>
99       implicit val s = subscriber.scheduler
100       val observer = subscriber.observer
101 
102       source.unsafeSubscribe(new Observer[T] {
103         def onNext(elem: T) = observer.onNext(elem)
104         def onError(ex: Throwable) = observer.onError(ex)
105         def onComplete() = observer.onError(error)
106       })
107     }
108 
109   /**
110    * Implements [[Observable.isEmpty]].
111    */
112   def isEmpty[T](source: Observable[T]): Observable[Boolean] =
113     Observable.create[Boolean] { subscriber =>
114       implicit val s = subscriber.scheduler
115       val o = subscriber.observer
116 
117       source.unsafeSubscribe(new Observer[T] {
118         def onNext(elem: T): Future[Ack] = {
119           o.onNext(false).onContinueSignalComplete(o)
120           Cancel
121         }
122 
123         def onError(ex: Throwable): Unit =
124           o.onError(ex)
125 
126         def onComplete(): Unit = {
127           // if we get here, it means that `onNext` never happened
128           o.onNext(true).onContinueSignalComplete(o)
129         }
130       })
131     }
