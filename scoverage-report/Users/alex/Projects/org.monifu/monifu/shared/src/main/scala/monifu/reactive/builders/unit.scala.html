24 object unit {

26    * Implementation for [[Observable.unit]].
27    */
28   def one[A](elem: A): Observable[A] =
29     Observable.create { s =>
30       s.observer.onNext(elem)
31         .onContinueSignalComplete(s.observer)(s.scheduler)
32     }
33 
34   /**
35    * Implementation for [[Observable.unitDelayed]].
36    */
37   def oneDelayed[A](delay: FiniteDuration, elem: A): Observable[A] =
38     Observable.create { s =>
39       s.scheduler.scheduleOnce(delay) {
40         s.observer.onNext(elem)
41           .onContinueSignalComplete(s.observer)(s.scheduler)
42       }
43     }
44 
45   /**
46    * Implementation for [[Observable.empty]].
47    */
48   def empty: Observable[Nothing] =

52    * Implementation for [[Observable.error]].
53    */

56 
57   /**
58    * Implementation for [[Observable.never]].
59    */
60   def never: Observable[Nothing] =
61     Observable.create { _ => () }
