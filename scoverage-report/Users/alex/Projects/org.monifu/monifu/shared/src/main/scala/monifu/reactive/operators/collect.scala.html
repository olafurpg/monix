24 object collect {
25   /**
26    * Implementation for [[Observable.collect]].
27    */
28   def apply[T,U](source: Observable[T])(pf: PartialFunction[T,U]): Observable[U] = {
29     Observable.create { subscriber =>
30       implicit val s = subscriber.scheduler
31       val observer = subscriber.observer
32 
33       source.unsafeSubscribe(new Observer[T] {
34         def onNext(elem: T) = {
35           // See Section 6.4. in the Rx Design Guidelines:
36           // Protect calls to user code from within an operator
37           var streamError = true
38           try {
39             if (pf.isDefinedAt(elem)) {
40               val next = pf(elem)
41               streamError = false
42               observer.onNext(next)
43             }
44             else
45               Continue
46           }
47           catch {
48             case NonFatal(ex) =>
49               if (streamError) { observer.onError(ex); Cancel } else Future.failed(ex)
50           }
51         }
52 
53         def onError(ex: Throwable) =
54           observer.onError(ex)
55 
56         def onComplete() =
57           observer.onComplete()
58       })
59     }
60   }
